118 function! MyFoldText() " {{{
119   let line = getline(v:foldstart)
120 
121   let nucolwidth = &fdc + &number * &numberwidth
122   let windowwidth = winwidth(0) - nucolwidth - 3
123   let foldedlinecount = v:foldend - v:foldstart
124 
125   " expand tabs into spaces
126   let onetab = strpart('          ', 0, &tabstop)
127   let line = substitute(line, '\t', onetab, 'g')
128 
129   let line = strpart(line, 0, windowwidth - 2 -len(foldedlinecount))
130   let fillcharcount = windowwidth - len(line) - len(foldedlinecount)
131   return line . '…' . repeat(" ",fillcharcount) . foldedlinecount . '…' . ' '
132 endfunction " }}}
133 set foldtext=MyFoldText()
